			NOT = { is_in_faction_with = PRC }
		}
		available = {

			NOT = { is_in_faction_with = SOV }
			NOT = { is_in_faction_with = PRC }
		}
		
		complete_effect = {
            add_ideas = non_aligned_movement_member
		}
		
		icon = politics
		
		cost = 100
		
		ai_will_do = {
			base = 0
		}
	}
	
	leave_non_aligned_movement = {

		}

		cost = 50

			base = 0
		}
	}
